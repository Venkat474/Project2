   //enter below lines below the 2nd fi
M2_HOME=/opt/maven
M2=/opt/maven/bin
JAVA_HOME=/usr/lib/jvm/java-11-openjdk-11.0.19.0.7-1.amzn2.0.1.x86_64   (Right click on the tab & Duplicate tab -># sudo su -># find / -name java-11*->u can see file here copy)
PATH=$PATH:$HOME/bin:$JAVA_HOME:$M2_HOME:$M2          :wq
➡️# echo $PATH                             (Still it is not showing the path for the maven & the java)
➡️# source .bash_profile                   (which will save the changes made to .bash_profile)
➡️# echo $PATH                             (Now it is showing the complete path for maven also & binary folder of maven also)
➡️# mvn -v                                 
(before we see inside bin folder only we can see maven but now i am not in maven then also i am going to see maven this is important change we have done, now i can run the maven anywhere on the server)
